public class ASTIntegerLiteral extends ASTExpression {

    public ASTIntegerLiteral(int value) {
        value_ = value;
    }
    public int value() {
        return value_;
    }

    public void setValue(int value) {
        value_ = value;
    }

    public Object accept(ASTVisitor V) {
        return V.VisitIntegerLiteral(this);
    }

    private int value_;
}
